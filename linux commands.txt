# cat (to open file)
  cat > <filename> (override data)
  cat >> <filename> (concatenate data)

# ls  (list)

# tail (to open last log files ,by default it open  last 10lines)
  tail -f <filename> (to check running logs i.e latest logs printed)
  tail -n 100 <filename> (to open last 100 log files)

# head (to open first log files, by default it open first 10 lines)
  head -f <filename> (to check running logs i.e latest logs printed)
  head -n 100 <filename> (to open first 100 log files)  

# htop (to check pid,cpu and memory of running processes in better way)

# free command (to check system memory )
  free -g (to show free memory in gb)
  free -m (to show free memory in mb)
